    链表¶

    概念¶

    链表是一种物理存储单元上非连续、非顺序的存储结构，数据元素的逻辑顺序是通过链表中的指针链接次序实现的。链表由一系列结点（链表中每一个元素称为结点）组成，结点可以在运行时动态生成。每个结点包括两个部分：一个是存储数据元素的数据域，另一个是存储下一个结点地址的指针域。 相比于线性表顺序结构，操作复杂。由于不必须按顺序存储，链表在插入的时候可以达到O(1)的复杂度，比另一种线性表顺序表快得多，但是查找一个节点或者访问特定编号的节点则需要O(n)的时间，而线性表和顺序表相应的时间复杂度分别是O(logn)和O(1)。

    优缺点分类¶

    链表是一种物理存储单元上非连续、非顺序的存储结构，数据元素的逻辑顺序是通过链表中的指针链接次序实现的。链表由一系列结点（链表中每一个元素称为结点）组成，结点可以在运行时动态生成。每个结点包括两个部分：一个是存储数据元素的数据域，另一个是存储下一个结点地址的指针域。 相比于线性表顺序结构，操作复杂。由于不必须按顺序存储，链表在插入的时候可以达到O(1)的复杂度，比另一种线性表顺序表快得多，但是查找一个节点或者访问特定编号的节点则需要O(n)的时间，而线性表和顺序表相应的时间复杂度分别是O(logn)和O(1)。

    一般分为单向链表,双向链表和环形链表

    单链表¶

    一般为了方便对单链表进行增删改查,一般会设置一个头节点,用来标识链表头,这个节点本身不存放数据.

    代码实现¶

    package main

import "fmt"

type Object interface{}

//定义一个Node
type Node struct {
    Data Object //定义数据域
    Next *Node  //定义指针域
}

//头节点
type List struct {
    headNode *Node
}

//判断是否为空的单链表
func (this *List) IsEmpty() bool {
    if this.headNode == nil {
        return true
    } else {
        return false
    }
}

//获取列表长度
func (this *List) Length() int {
    //获取链表头节点
    cur := this.headNode
    //定义一个计数器,初始值为0
    count := 0

    for cur != nil {
        count++
        cur = cur.Next //根据指针域找到下个数据
    }
    return count
}

//从链表头部添加元素
func (this *List) Add(data Object) *Node {
    node := &Node{Data: data}
    node.Next = this.headNode
    this.headNode = node
    return node
}

//从链表尾部添加元素
func (this *List) Append(data Object) {
    //创建一个新元素,通过传入data参数进行数据域的赋值
    node := &Node{Data: data}
    if this.IsEmpty() { //如果头节点为空,则直接将新元素作为头节点
        this.headNode = node
    } else { //链表不为空
        cur := this.headNode  //定义变量用于存储头节点
        for cur.Next != nil { //判断是否是尾节点 = nil 标识为尾节点
            cur = cur.Next //链表进行位移,通过cur获取到尾节点
        }
        cur.Next = node // 此时cur为尾节点,将其地址域指向新创建的节点
    }
}

//链表指定位置添加元素,index指下标,this.headNode的index = 0
func (this *List) Insert(index int, data Object) {
    if index < 0 { //如果index<0,头部插入
        this.Add(data)
    } else if index > this.Length() {
        this.Append(data) //index大于链表长度,进行尾部插入
    } else {
        pre := this.headNode
        count := 0
        for count < (index - 1) { //用于控制位移的链表数目
            pre = pre.Next
            count++
        }
        //当循环结束后,pre指向index-1的位置
        node := &Node{Data: data} //创建新的链表元素
        node.Next = pre.Next      //将新的链表元素地址指向上一个链表地址域存储的地址(也就是新元素现在的跟随元素)\
        pre.Next = node           //上一个链表地址域存储的地址指向新的元素地址
    }
}

//删除指定值元素
func (this *List) RemoveAtValue(data Object) {
    pre := this.headNode  //定义变量pre,存储头节点
    if pre.Data == data { //如果该节点的数据域的值为data
        this.headNode = pre.Next // 如果是就删除头节点,将第二个节点改为新的头节点
    } else { //如果删除的不是头节点
        for pre.Next != nil { //循环判断是否遍历到最后一个节点,如果不是,进行判断
            if pre.Next.Data == data { //如果pre.Next的节点的数据等于data,那么就删除该节点,pre.Next指向该节点之后的节点地址
                pre.Next = pre.Next.Next
            } else { //如果pre.Next的节点数据不等于data,就及逆行节点位移,循环判断
                pre = pre.Next
            }
        }
    }
}

//删除指定位置的元素
func (this *List) RemoveAtIndex(index int) {
    pre := this.headNode
    if index <= 0 { //如果index为小于等于0,则删除头节点
        this.headNode = pre.Next //将第二个节点作为头节点
    } else if index > this.Length() { // 如果传入的参数大于链表长度,则报错,退出程序
        fmt.Println("超出链表长度")
        return
    } else {
        count := 0
        for count != (index-1) && pre.Next != nil { //开始遍历,如果index=1,删除并跳出循环,如果index>1,进行链表的移位,然后删除
            count++
            pre = pre.Next
        }
        pre.Next = pre.Next.Next
    }
}

//查看链表是否包含某个元素
func (this *List) Contain(data Object) bool {
    cur := this.headNode
    for cur != nil {
        if cur.Data == data {
            return true
        }
        cur = cur.Next
    }
    return false
}

//遍历链表所有节点
func (this *List) ShowList() {
    if !this.IsEmpty() {
        cur := this.headNode

        for {
            fmt.Printf("\t%v", cur.Data)
            if cur.Next != nil {
                cur = cur.Next
            } else {
                break
            }
        }
        fmt.Println()
    }
}

func main() {
    //初始化链表List
    list := List{}
    //测试为空
    fmt.Printf("初始链表:%t\n", list.IsEmpty())
    //尾部追加1,2,3,a,b,c
    list.Append(1)
    list.Append(2)
    list.Append(3)
    list.Append("a")
    list.Append("b")
    list.Append("c")
    fmt.Printf("链表长度为%d\n", list.Length())
    //打印链表
    fmt.Println("链表为:")
    list.ShowList()
    //insert插入
    list.Insert(5, "z")
    fmt.Println("链表为:")
    list.ShowList()
    //成员是否存在
    isContain := list.Contain("z")
    fmt.Printf("isContain[z]:%t\n", isContain)
    //删除指定值的成员
    list.RemoveAtValue("z")
    fmt.Println("链表为:")
    list.ShowList()
    //删除指定索引的成员
    list.RemoveAtIndex(3)
    fmt.Println("链表为:")
    list.ShowList()
}

    环形双链表¶

    包含prev和next两个指针域,分别指向前后两个元素

    表中最后一个结点的指针域指向头结点，整个链表形成一个环。

    代码实现¶
    package main

import (
    "fmt"
)

//双向环形链表结构体
type CircleLink struct {
    Id   int         //节点索引
    Data interface{} //数据域
    prev *CircleLink //prev域
    next *CircleLink //next域
}

//初始化头节点元素,头节点的id初始化为1,获取到的结构体实例作为头节点存在
func InitHeadNode(data interface{}) *CircleLink {
    return &CircleLink{
        Id:   1,
        Data: data,
    }
}

//重置头元素
func (head *CircleLink) ResetHeadNode(data interface{}) {
    if head.Id == 0 {
        head.Id = 1
    }
    head.Data = data
}

//判断当前头部是否为空
func (head *CircleLink) IsHeadEmpty() bool {
    return head.next == nil && head.prev == nil
}

//判断当前是否为空链表
func (head *CircleLink) IsEmptyLink() bool {
    return head.Data == nil && head.next == nil && head.prev == nil && head.Id == 0
}

//末尾元素
func (head *CircleLink) GetLastNode() *CircleLink {
    //如果只有一个头元素直接返回
    currentNode := head
    if !head.IsHeadEmpty() {
        //如果不为空,遍历到最后一个元素
        for {
            if currentNode.next == head { //表示找到
                break
            }
            currentNode = currentNode.next //让当前节点向下继续
        }
    }
    return currentNode
}

//从头节点按顺序插入双向链表节点
func (head *CircleLink) AddNode(newNode *CircleLink) {
    //如果只有一个元素则初始化next和prev指针形成双向环形链表
    if head.IsHeadEmpty() {
        //头节点则让next指针指向newNode
        head.next = newNode
        //头节点则让prev指向nil
        head.prev = newNode
        //让newNode的prev和next指针都指向head
        newNode.prev = head
        newNode.next = head
        return
    }
    //如果环形已经形成则按顺序添加节点,将当前节点指向头部
    currentNode := head
    //flag表示是否应该进行中间插入
    flag := false //默认添加到最后
    for {
        if currentNode == head.prev {
            //已经是最后一个元素则表示添加到末尾
            break
        } else if currentNode.next.Id > newNode.Id {
            //fmt.Println("如果当前节点的next大于newNode.id则找到了添加的位置")
            //如果当前节点的next大于newNode.id则找到了添加的位置
            flag = true
            break
        } else if currentNode.next.Id == newNode.Id {
            fmt.Println("error:id already exists")
            return
        }
        //当前节点继续前进
        currentNode = currentNode.next
    }

    if flag {
        //在中间添加
        newNode.next = currentNode.next
        newNode.prev = currentNode
        currentNode.next.prev = newNode
        currentNode.next = newNode
    } else {
        //尾部添加
        newNode.prev = currentNode
        newNode.next = currentNode.next
        currentNode.next = newNode
        head.prev = newNode
    }
}

//按照id查找节点
func (head *CircleLink) FindNodeById(id int) (*CircleLink, bool) {
    if head.IsHeadEmpty() && head.Id == id {
        return head, true
    } else if head.IsHeadEmpty() && head.Id != id {
        return &CircleLink{}, false
    }
    //头部非空则遍历查找
    currentNode := head
    flag := false
    for {
        if currentNode.Id == id {
            flag = true
            break
        }
        if currentNode == head.prev { //找到最后,表示不存在
            break
        }
        currentNode = currentNode.next
    }
    if !flag {
        return &CircleLink{}, false
    }
    return currentNode, true
}

//删除指定id的节点
func (head *CircleLink) DeleteNodeById(id int) bool {
    if head.IsEmptyLink() {
        fmt.Println("无法删除空链表")
        return false
    }
    node, boolen := head.FindNodeById(id)

    if boolen {
        //删除的头部元素
        if node == head {
            //处理只有一个头部元素的情况,则把头元素进行id初始化设置并把Data设置为nil
            if head.IsHeadEmpty() {
                head.prev = nil
                head.next = nil
                head.Data = nil
                head.Id = 0
                return true
            }
            //如果只有头节点和末尾节点两个元素
            if head.next.next == head {
                nextNode := head.next
                head.Id = nextNode.Id
                head.Data = nextNode.Data
                head.prev = nil
                head.next = nil
                return true
            }
            //如果大于两个节点则移动下一个节点作为头节点
            nextNodeTmp := head.next
            head.Data = nextNodeTmp.Data
            head.Id = nextNodeTmp.Id
            head.next = nextNodeTmp.next
            nextNodeTmp.next.prev = head
            return true
        }
        //删除的元素是末尾元素
        if node == head.GetLastNode() {
            //如果只有两个元素
            fmt.Println("最后的节点:", node.Id, node.Data, node.next.Data, node.prev.Data)
            if node.prev == head && node.next == head {
                //如果只有两个元素直接在head节点中断连接
                head.prev = nil
                head.next = nil
                return true
            }
            head.prev = node.prev
            node.prev.next = head
            return true
        }
        //删除的元素并非头节点也非末尾节点
        node.prev.next = node.next
        node.next.prev = node.prev
        return true
    } else {
        fmt.Println("找不到要删除的节点")
    }
    return boolen
}

//根据id修改data数据
func (head *CircleLink) ModifyById(id int, data interface{}) bool {
    node, boolean := head.FindNodeById(id)
    if boolean {
        node.Data = data
    } else {
        fmt.Println("找不到修改的id")
    }
    return boolean
}

//遍历整个环形链表
func (head *CircleLink) FetchAll() {
    if head.IsEmptyLink() {
        fmt.Println("无法遍历空链表")
        return
    }
    if head.IsHeadEmpty() {
        fmt.Println("[", head.Id, head.Data, "next:", head.next, "prev:", head.prev, "]")
        return
    }
    currentNode := head
    for {
        fmt.Println("[", currentNode.Id, currentNode.Data, "next:", currentNode.next.Data, " perv:", currentNode.prev.Data, "]")
        if currentNode == head.prev {
            break
        }
        currentNode = currentNode.next
    }
}

//测试
func main() {
    var t1 interface{} = "张三"
    var t2 interface{} = "李四"
    var t3 interface{} = "王五"
    var t4 interface{} = "赵六"
    var t5 interface{} = "白七"

    linkNode := InitHeadNode(t1)
    node3 := &CircleLink{
        Id:   3,
        Data: t2,
    }
    node2 := &CircleLink{
        Id:   2,
        Data: t3,
    }
    node5 := &CircleLink{
        Id:   5,
        Data: t4,
    }
    node4 := &CircleLink{
        Id:   4,
        Data: t5,
    }
    linkNode.AddNode(node3)
    linkNode.AddNode(node2)
    linkNode.AddNode(node5)
    linkNode.AddNode(node4)
    linkNode.DeleteNodeById(4)
    linkNode.FetchAll()
    var t6 interface{} = "老大"
    updateData := t6
    linkNode.ModifyById(3, updateData)
    linkNode.FetchAll()
}
