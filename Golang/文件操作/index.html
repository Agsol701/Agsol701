    文件操作¶

    流:数据在数据源(文件)和程序(内存)之间经历的路径

    输入流:数据从数据源(文件)到程序(内存)的路径 -- 读文件

    输出流:数据从程序(内存)到数据源(文件)的路径 --写文件

    golang中的文件操作¶

    通过os包中的os.File结构体来实现文件操作

    File表示一个打开的文件

    常用的文件操作函数和方法¶

    打开文件和关闭文件¶

    func Open

    func Open(name string) (file *File,err error)

    Open打开一个文件用于读取,如果操作成功,返回的文件对象的方法可用于读取数据;对应的文件描述符具有O_RDONLY模式.如果出错,错误底层类型为*PathError

    func(*File)Close

    func (f *File) Close() error

    Clsoe关闭文件f,使文件不能用于读写,返回可能出现的错误.

    func main(){
    //打开文件
    //下方file 可以称为file对象,file句柄或file指针
    file,err := os.Open("C:/Users/AGSOL/Desktop/text.txt")
    if err != nil {
        fmt.Println("open file err = ",err)
    }
    //输出文件
    fmt.Println(file)  //&{0xc00008a780}
    fmt.Println(*file) //{0xc00008a780}

    //关闭文件
    err = file.Close()
    if err != nil {
        fmt.Println("close file error :",err)
    }
}

    读文件操作¶

    读取文件内容并显示在终端(带缓冲区的方式)¶

    适用于文件比较大情况

    os.Open, file.Close, bufio.NewReader(), reader.ReadString函数和方法

    package main

import (
    "bufio"
    "fmt"
    "io"
    "os"
)

func main() {
    //打开文件
    //下方file 可以称为file对象,file句柄或file指针
    file, err := os.Open("C:/Users/AGSOL/Desktop/text.txt")
    if err != nil {
        fmt.Println("open file err = ", err)
    }

    //当函数退出时,关闭file,defer是在函数结束前执行的语句
    //要及时关闭file句柄,否则出现内存泄漏
    defer file.Close()

    //创建一个带缓冲的*Reader
    /*NewReader默认缓冲区大小为4096
    const (
        defaultBufSize = 4096
    )
    */
    reader := bufio.NewReader(file)
    //循环读取文件内容
    for {
        str, err := reader.ReadString('\n') //读到一个\n就结束
        fmt.Print(str)
        if err == io.EOF { //io.EOF表示读到了文件的末尾
            break
        }
        //输出内容
    }
    fmt.Println("read over")
}
>>>
111111111
222222222
333333333read over

    写文件操作¶

    写文件要借助打开文件os.OpenFile方法,其中打开方式的常量如下,这些常量可以通过|来组合使用

    const (
    // Exactly one of O_RDONLY, O_WRONLY, or O_RDWR must be specified.
    O_RDONLY int = syscall.O_RDONLY // open the file read-only.
    O_WRONLY int = syscall.O_WRONLY // open the file write-only.
    O_RDWR   int = syscall.O_RDWR   // open the file read-write.
    // The remaining values may be or'ed in to control behavior.
    O_APPEND int = syscall.O_APPEND // append data to the file when writing.
    O_CREATE int = syscall.O_CREAT  // create a new file if none exists.
    O_EXCL   int = syscall.O_EXCL   // used with O_CREATE, file must not exist.
    O_SYNC   int = syscall.O_SYNC   // open for synchronous I/O.
    O_TRUNC  int = syscall.O_TRUNC  // truncate regular writable file when opened.
)

    通过os.OpenFile函数,此函数接收三个参数,第一是文件路径,第二个是打开模式(可以组合),第三个参数是权限控制,windows操作系统下无效,在linux系统下使用.

    如果操作成功,返回的文件对象可用于I/O.如果出错,错误底层类型为*PathError.

    举例¶
    1. 新建一个文件,写入5句hello world
    package main

import (
    "bufio"
    "fmt"
    "os"
)

func main() {
    //创建一个新文件,写入五句hello world
    //创建文件
    filePath := "C:/Users/AGSOL/Desktop/abc.txt"
    file, err := os.OpenFile(filePath, os.O_WRONLY|os.O_CREATE, 0666)
    if err != nil {
        fmt.Println("open file error:", err)
        return
    }

    //及时关闭file句柄,避免内存泄漏
    defer file.Close()

    //写入内容
    str := "hello world\n"
    //使用带缓存的*Writer
    writer := bufio.NewWriter(file)
    for i := 0; i < 5; i++ {
        writer.WriteString(str)
    }

    //因为writer是带缓存的,因此要使用flush刷到硬盘上
    writer.Flush()
}

    判断文件或文件夹是否存在¶

    goalng判断文件或文件夹是否存在,使用os.Stat()函数返回的错误值进行判断

    1. 如果返回的错误为nil,说明文件加存在
    2. 如果返回的错误类型使用os.IsNotExist()判断为true,说明文件或文件夹不存在
    3. 如果返回的错误结果为其他类型,则不确定是否存在

    通过一下函数来判断文件或文件夹是否存在

    func PathExists(path string) (bool,error) {
    _,err ;= os.Stat(path)
    if err == nil { //文件或目录存在
        return true,nil
    }
    if os.IsNotExist(err) {
        return false,nil
    }
    return false,err
}

    文件的拷贝¶

    使用io包中的拷贝函数:func Copy(dst Writer, src Reader) (written int64, err error)

    可以拷贝文本,音频 ,视频

    package main

import (
    "bufio"
    "fmt"
    "io"
    "os"
)

//自己写一个函数,接收两个文件路径,srcFileName,dstFileName
func CopyFile(dstFileName, srcFileName string) (written int64, err error) {
    srcFile, err := os.Open(srcFileName)
    if err != nil {
        fmt.Println("open file error", err)
    }

    defer srcFile.Close()

    //从srcfile获取Reader
    reader := bufio.NewReader(srcFile)
    //打开dstFileName
    dstFile, err := os.OpenFile(dstFileName, os.O_WRONLY|os.O_CREATE, 0666)
    if err != nil {
        fmt.Println("open file error ", err)
    }
    //通过dstFile获取writer
    writer := bufio.NewWriter(dstFile)

    defer dstFile.Close()

    return io.Copy(writer, reader)
}
func main() {
    //将C:/Users/AGSOL/Desktop/8c42fca6bb074d68d10d9bea29d3ff6.jpg
    //拷贝到C:/Users/AGSOL/Desktop/video
    srcFile := "C:/Users/AGSOL/Desktop/8c42fca6bb074d68d10d9bea29d3ff6.jpg"
    dstFile := "C:/Users/AGSOL/Desktop/video/abc.jpg"
    _, err := CopyFile(dstFile, srcFile)
    if err == nil {
        fmt.Println("ok")
    } else {
        fmt.Println("copy error ", err)
    }
}
    统计英文,数字,空格和其他字符数量¶

    统计一个文件中含有的英文,数字,空格及其他字符数量

    package main

import (
    "bufio"
    "fmt"
    "io"
    "os"
)

//定义一个结构体,用于保存统计的结果
type CharCount struct {
    ChCount    int //记录英文个数
    NumCount   int //记录数字个数
    SpaceCount int //记录空格个数
    OtherCount int //纪录其他字符个数
}

func main() {
    //思路:打开一个文件创建一个reader
    //按行读取,统计有多少个字符
    //然后将结果保存到一个结构体中
    fileName := "C:/Users/AGSOL/Desktop/text.txt"
    file, err := os.Open(fileName)
    if err != nil {
        fmt.Println("open file err", err)
    }

    defer file.Close()
    //实例化一个结构体实例
    var count CharCount
    //创建一个reader
    reader := bufio.NewReader(file)
    //循环读取文件内容
    for {
        str, err := reader.ReadString('\n')
        if err == io.EOF {
            break
        }
        //为了兼容中文
        str = []rune(str)
        //遍历str进行统计
        for _, v := range str {
            // fmt.Println(v)
            switch {
            case v >= 'a' && v <= 'z':
                //count.ChCount ++
                fallthrough //穿透t
            case v >= 'A' && v <= 'Z':
                count.ChCount++
            case v == ' ' || v == '\t':
                count.SpaceCount++
            case v >= '0' && v <= '9':
                count.NumCount++
            default:
                count.OtherCount++
            }

        }
    }

    fmt.Println(count)
}
