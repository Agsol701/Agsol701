    排序¶

    排序是指将一系列无序数组整理成有序的数组的过程.

    冒泡排序¶

    时间复杂度:最坏O(n^2),最优O(n)

    算法过程¶

    遍历整个数组,比较相邻元素,将大的数字向后走

    外层循环控制循环次数

    内层循环控制比较哪两个元素

    优化¶

    如果一次比较来下没有元素进行交换,就说明已经排好序

    python实现¶
    def bubble_sort(lst):
    for i in range(len(lst)-1):
        flag = True
        for j in range(len(lst)-1-i):
            if lst[j] < lst[j+1]:
                lst[j],lst[j+1] = lst[j+1],lst[j]
                flag = False

        if flag:
            return
    golang实现¶
    package main

import "fmt"

func BubbleSort (arr []int) []int {
    for i:=0; i<len(arr)-1; i++ {
        flag := true
        for j:=0; j<len(arr)-i-1; j++{
            if arr[j] < arr[j+1] {
                arr[j],arr[j+1] = arr[j+1],arr[j]
                flag = false
            }
        }
        if flag {
            return arr
        }
    }
    return arr
}

func main()  {
    arr := []int{1,9,3,6,4,8,0,2,5,7}
    fmt.Println(arr)
    BubbleSort(arr)
    fmt.Println(arr)
}

    选择排序¶

    时间复杂度:O(n^2)

    算法过程¶

    第一次从 1 到 n-1 个元素中选择一个最小数据与位置 0 的数据交换 第二次从 2 到 n -1 个元素中选择一个最小数据与位置 1 的数据交换 第三次从 3 到 n -1 个元素中选择一个最小数据与位置 2 的数据交换

    优化¶

    一次设定一个最大值和最小值,两侧同时向中间逼近

    python实现¶
    def select_sort(lst):
    for i in range(len(lst)):
        min_index = i
        for j in range(i+1,len(lst)):
            if lst[j] < lst[min_index]:
                lst[j],lst[min_index] = lst[min_index],lst[j]
    golang实现¶
    package main

import "fmt"

func SelectSort (arr []int) []int {
    for i:=0; i<len(arr); i ++ {
        min_index := i
        for j := i+1; j<len(arr); j++ {
            if arr[j] < arr[min_index] {
                arr[j],arr[min_index] = arr[min_index],arr[j]
            }
        }
    }
    return arr
}

func main()  {
    arr := []int{1,9,3,6,4,8,0,2,5,7}
    fmt.Println(arr)
    SelectSort(arr)
    fmt.Println(arr)
}

    快速排序¶

    时间复杂度O(nlogn)

    只经过一次遍历

    算法¶

    首先将整组的中位数选出来,

    再将数组从中位数两侧分成两部分

    递归求中位数,最后得到整个排序数组.

    递归退出条件:左等于右

    python实现¶
    def partition(lst,left,right):
    tmp = lst[left]
    while left < right:
        while left < right and lst[right]>=tmp:
            right=right - 1
        #当找到边界时,替换位置
        lst[left] = lst[right]

        while left < right and lst[left]<=tmp:
            left=left + 1
        lst[right] = lst[left]
    #此时left=right
    lst[right] = tmp

    return left

def quick_sort(lst,left,right):
    #注意left和right,从0开始到len(lst)-1结束
    if left < right:
        mid = partition(lst,left,right)

        quick_sort(lst,left,mid-1)
        quick_sort(lst,mid+1,right)
    golang实现¶
    package main

import "fmt"

func partition(arr []int,left,right int) int  {
    tmp := arr[left]
    for left < right {
        for left < right && arr[right] >= tmp {
            right -= 1
        }
        arr[left] = arr[right]
        for left < right && arr[left] <= tmp {
            left += 1
        }
        arr[right] = arr[left]
    }
    arr[left] = tmp
    return left
}

func QuickSort(arr []int, left, right int)  {
    if left < right {
        mid := partition(arr,left,right)

        QuickSort(arr,left,mid-1)
        QuickSort(arr,mid+1,right)
    }
}

func main()  {
    arr := []int{1,9,3,6,4,8,0,2,5,7}
    fmt.Println(arr)
    //BubbleSort(arr)
    //SelectSort(arr)
    QuickSort(arr,0,len(arr)-1)
    fmt.Println(arr)
}

    计数排序¶

    它**适用于一定范围的整数排序**

    算法¶

    以（数列最大值和最小值的差+1）作为统计数组的长度，数列的最小值作为一个偏移量，生成一个计数数组且此数组初始值为0.

    遍历源数组,每出现一个就在对应索引上加1.

    再将统计数组中的索引(转换为源数据)和对应出现次数,就可以精准排序.

    def countingsort(lst):
    min_num = min(lst)
    max_num = max(lst)
    lst_len = max_num - min_num
    count_lst = [0] * (lst_len+1)
    for i in lst:
        count_lst[i-min_num] += 1
        # print(count_lst)
    lst.clear()
    for index,count in enumerate(count_lst):
        l = [index+min_num] * count
        lst += l
